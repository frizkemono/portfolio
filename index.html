local function isAreaOccupied(holder, col, row, w, h, ignoreBtn)
	for _, child in ipairs(holder:GetChildren()) do
		if child:IsA("ImageButton") and child ~= ignoreBtn then
			local cx = child:GetAttribute("Col")
			local cy = child:GetAttribute("Row")
			local cw, ch = itemDims(child.Name)
			if cx and cy then
				if col < cx + cw and col + w > cx and row < cy + ch and row + h > cy then
					return true
				end
			end
		end
	end
	return false
end

local function isAreaWithinValidSlots(col, row, w, h)
	for r = row, row + h - 1 do
		for c = col, col + w - 1 do
			if not isSlotIndexValid(c, r) then
				return false
			end
		end
	end
	return true
end

local function findFreeSlot(holder, w, h)
	local rows = holder == InvHolder and getTotalRows() or 6
	local cols = GRID_W
	for y = 1, rows - h + 1 do
		for x = 1, cols - w + 1 do
			local valid = true
			if holder == InvHolder then
				valid = isAreaWithinValidSlots(x, y, w, h)
			end
			if valid and not isAreaOccupied(holder, x, y, w, h) then
				return x, y
			end
		end
	end
	return nil, nil
end

local function getGhostCoord(holder, w, h)
	if cellSize == 0 then return nil, nil end
	local m = mousePos()
	local pos = holder.AbsolutePosition
	local cs = cellSize + PADDING
	local rows = holder == InvHolder and getTotalRows() or 6
	local col = math.floor((m.X - pos.X - PADDING) / cs - (w - 1) / 2) + 1
	local row = math.floor((m.Y - pos.Y - PADDING) / cs - (h - 1) / 2) + 1
	col = math.clamp(col, 1, GRID_W - w + 1)
	row = math.clamp(row, 1, rows - h + 1)
	return col, row
end

local ReplicatedStorage = game:GetService("ReplicatedStorage")

local moduleScript = script
local container = moduleScript.Parent

local remoteEvent = nil
local remoteFunction = nil
local fastSpinResultCallbacks = {}
local currentUUID = nil
local clientConnected = false

local function findRemote(containerObj, className, name)
	if not containerObj then return nil end
	local direct = containerObj:FindFirstChild(name)
	if direct and direct.ClassName == className then return direct end
	for _, child in ipairs(containerObj:GetDescendants()) do
		if child.Name == name and child.ClassName == className then return child end
	end
	return nil
end

local function resolveContainer()
	if container then return container end
	local sutfFolder = ReplicatedStorage:FindFirstChild("sutf")
	return sutfFolder or moduleScript
end

local function getRemoteEvent()
	if remoteEvent then return remoteEvent end
	local root = resolveContainer()
	remoteEvent = findRemote(root, "RemoteEvent", "RemoteEvent")
	if not remoteEvent and root ~= moduleScript then
		remoteEvent = findRemote(moduleScript, "RemoteEvent", "RemoteEvent")
	end
	if not remoteEvent then error("re not found.") end
	return remoteEvent
end

local function getRemoteFunction()
	if remoteFunction then return remoteFunction end
	local root = resolveContainer()
	remoteFunction = findRemote(root, "RemoteFunction", "RemoteFunction")
	if not remoteFunction and root ~= moduleScript then
		remoteFunction = findRemote(moduleScript, "RemoteFunction", "RemoteFunction")
	end
	return remoteFunction
end

local module = {}

function module.getUUID() return currentUUID end

function module.invoke(action, data)
	return getRemoteFunction():InvokeServer(action, data)
end

function module.fire(action, data)
	getRemoteEvent():FireServer(action, data)
end

return module
